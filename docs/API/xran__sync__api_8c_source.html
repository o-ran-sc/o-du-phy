66  return 1;
67  }
68 
69  struct dirent *entry = NULL;

71  long pid = atol(entry->d_name);
72  if (0 == pid)
73  continue;
74  sprintf(full_path, "%s/%ld/%s", PROC_DIR, pid, COMM_FILE);
75  FILE *proc_name_file = fopen(full_path, "r");
76  if (NULL == proc_name_file)
77  continue;

80  res = 0;

82  break;
83  }
84  fclose(proc_name_file);
85  }
